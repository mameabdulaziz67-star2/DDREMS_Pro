STEP 2: Check Data Integrity
  Command: mysql -u root -p ddrems
  
  Then run:
    SELECT COUNT(*) FROM messages;
    SELECT COUNT(*) FROM message_recipients;
    SELECT COUNT(*) FROM notifications;

  Expected: Counts should match (or be reasonable)

STEP 3: Check for Orphaned Messages
  Command: mysql -u root -p ddrems
  
  Then run:
    SELECT COUNT(*) FROM messages m
    WHERE m.sender_id NOT IN (SELECT id FROM users)
    OR (m.receiver_id IS NOT NULL AND m.receiver_id NOT IN (SELECT id FROM users));

  Expected: 0 (no orphaned messages)

STEP 4: Check Message Recipients
  Command: mysql -u root -p ddrems
  
  Then run:
    SELECT m.id, m.subject, COUNT(mr.user_id) as recipients
    FROM messages m
    LEFT JOIN message_recipients mr ON m.id = mr.message_id
    WHERE m.is_group = 1
    GROUP BY m.id;

  Expected: Group messages should have multiple recipients

📝 SAMPLE DATABASE QUERIES
═════════════════════════════════════════════════════════════════════════════

View all messages:
  SELECT m.id, u.name as sender, m.subject, m.is_group, m.is_read, m.created_at
  FROM messages m
  LEFT JOIN users u ON m.sender_id = u.id
  ORDER BY m.created_at DESC;

View group message recipients:
  SELECT m.id, m.subject, COUNT(mr.user_id) as recipients
  FROM messages m
  LEFT JOIN message_recipients mr ON m.id = mr.message_id
  WHERE m.is_group = 1
  GROUP BY m.id;

View unread messages for user:
  SELECT m.id, u.name as sender, m.subject, m.created_at
  FROM messages m
  LEFT JOIN users u ON m.sender_id = u.id
  WHERE m.receiver_id = [user_id] AND m.is_read = 0;

View message read status per recipient:
  SELECT mr.user_id, u.name, mr.is_read, mr.read_at
  FROM message_recipients mr
  JOIN users u ON mr.user_id = u.id
  WHERE mr.message_id = [message_id];
